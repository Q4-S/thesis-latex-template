\ifxelatex
% ---- XeLaTeX / LuaLaTeX block ----------------------------------------
% Compiler: xelatex or lualatex
% Overleaf:  Menu → Compiler → XeLaTeX

  \usepackage{fontspec}   % OpenType font selection (XeLaTeX/LuaLaTeX only)
  \usepackage{kotex}      % Korean support — uses fontspec internally
                          % Alternative: xeCJK (requires different font commands)

  % MAIN font — body text and paragraphs
  \setmainfont{Times New Roman}
  \setmainhangulfont{UnBatang}[
    % BoldFont     = NanumMyeongjo,  % UnBatang has native bold
    ItalicFont     = NanumMyeongjo,
    BoldItalicFont = NanumMyeongjo
  ]

  % SANS-SERIF font — headings, titles, captions   usage: \textsf{}
  \setsansfont{Arial}
  \setsanshangulfont{NanumGothic}

  % MONOSPACE font — code, URLs, terminal text      usage: \texttt{}
  \setmonofont{Courier New}
  \setmonohangulfont{NanumGothicCoding}

\else
% ---- pdfLaTeX block --------------------------------------------------
% Compiler: pdflatex
% Overleaf:  Menu → Compiler → pdfLaTeX  (default, works on free tier)
%
% Since \xelatexfalse means we are intentionally using pdfLaTeX,
% no engine detection needed — just load packages directly.
% No nested \if blocks here to avoid mismatched \fi errors.

  \usepackage[utf8]{inputenc}   % UTF-8 input encoding
  \usepackage[T1]{fontenc}      % output font encoding

  % Korean support via CJKutf8
  % Font families available for Korean:
  %   mj = 명조 Myungjoh  (serif,  closest to UnBatang)
  %   gr = 고딕 Gothic    (sans,   closest to NanumGothic)
  %   bm = 바탕 Batang
  \usepackage{CJKutf8}

  % English font substitutions (closest pdfLaTeX equivalents)
  \usepackage{mathptmx}              % Times New Roman (main/serif)
  \usepackage[scaled=0.92]{helvet}   % Helvetica/Arial  (sans-serif)
  \usepackage{courier}               % Courier New      (monospace)

  % \ko{} — convenience macro for inline Gothic Korean
  % usage: \ko{이것은 고딕체입니다}
  \newcommand{\ko}[1]{{\CJKfamily{gr}#1}}

\fi
% ---- End of compiler-dependent block ------------------------------------

% ============================================
% CHAPTER & SECTION TITLE FORMATTING
% ============================================

% Chapter heading style
\titleformat{\chapter}
  [block]
  {\sffamily\Huge\bfseries\centering}
  {\chaptertitlename\ \thechapter.}
  {0.5cm}
  {}

% Chapter numbering format shown in headings
\ifxelatex % Chapter numbering — Korean only in XeLaTeX mode
  % Example output: "제-1장  Chapter Title"
  \renewcommand{\chaptername}{제-}                     % prefix before number
  \renewcommand{\thechapter}{\arabic{chapter}장}       % suffix 장 after number
\else % Chapter numbering using pdflatex
  \renewcommand{\chaptername}{Chapter }
  \renewcommand{\thechapter}{\arabic{chapter}}
\fi

% Centering chapter titles and make it one line
\usepackage{xcolor}  % needed for \color{} in chapter title format
\titleformat
	{\chapter} % command to format
	[block] % shape: hang, display, block, frame etc
	{\bfseries \centering}  % format of whole number label + chapter title text
	{\large \color{black!80} \MakeUppercase{\chaptertitlename \thechapter} \\} % format of number label only, ex: add . or enter new line \\
	{0cm} % separation number label - chapter title text
	{\sffamily \Huge } % Format for the title text only

% Section / subsection heading styles
\titleformat{\section}{\sffamily\large\bfseries}{\thesection.}{1em}{}
\titleformat{\subsection}{\sffamily\normalsize\bfseries}{\thesubsection.}{1em}{}

% Vertical spacing around headings
% \titlespacing*{cmd}{left}{before}{after}
\titlespacing*{\chapter}   {0cm}{-1.5cm}{0.5cm}
\titlespacing*{\section}   {0cm}{1cm}{0.5cm}
\titlespacing*{\subsection}{0cm}{1cm}{0.5cm}
