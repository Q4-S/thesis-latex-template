% ============================================
% FONT & KOREAN SUPPORT  (compiler-dependent)
% ============================================

\ifxelatex
  % ---- XeLaTeX / LuaLaTeX block ----------------------------------------
  % Requires: compile with xelatex or lualatex
  % Overleaf: Menu → Compiler → XeLaTeX  (paid tier recommended for long docs)

  \usepackage{fontspec}         % OpenType font selection
  \usepackage{kotex}            % Korean/Hangeul support (uses fontspec internally)
  % Alternative: \usepackage{xeCJK}  — if you switch, update hangulfont commands below

  % --- MAIN FONTS (body text, paragraphs) ---
  \setmainfont{Times New Roman}
  \setmainhangulfont{UnBatang}[
    % BoldFont     = NanumMyeongjo,   % UnBatang supports bold natively
    ItalicFont     = NanumMyeongjo,
    BoldItalicFont = NanumMyeongjo
  ]

  % --- SANS-SERIF FONTS (headers, titles, captions)  \textsf{} ---
  \setsansfont{Arial}
  \setsanshangulfont{NanumGothic}

  % --- MONOSPACE FONTS (code, URLs, terminal text)   \texttt{} ---
  \setmonofont{Courier New}
  \setmonohangulfont{NanumGothicCoding}

\else
  % ---- pdfLaTeX block --------------------------------------------------
  % Requires: compile with pdflatex
  % Overleaf: Menu → Compiler → pdfLaTeX  (default, works on free tier)
  % Note: Korean text is wrapped with \ko{} defined below.
  %       For large Korean blocks use \begin{CJK}{UTF8}{mj}...\end{CJK}

  % Load inputenc only when actually running pdfLaTeX.
  % XeLaTeX/LuaLaTeX handle UTF-8 natively — loading inputenc under those
  % engines triggers: "inputenc package ignored with utf8 based engines."
  % The \iftex check below prevents that warning entirely.
  \usepackage{iftex}            % provides \ifPDFTeX, \ifXeTeX, \ifLuaTeX
  \ifPDFTeX
    \usepackage[utf8]{inputenc} % UTF-8 input encoding (pdfLaTeX only)
    \usepackage[T1]{fontenc}    % output font encoding  (pdfLaTeX only)
  \fi

  % Korean support via CJKutf8
  % Available CJK font families for Korean:
  %   mj  = Myungjoh (명조, serif — closest to UnBatang)
  %   gr  = Gothic   (고딕, sans-serif — closest to NanumGothic)
  %   bm  = Batang   (바탕)
  \usepackage{CJKutf8}

  % English font substitutions (visually close to XeLaTeX versions)
  \usepackage{mathptmx}               % Times New Roman equivalent  (main/serif)
  \usepackage[scaled=0.92]{helvet}    % Arial/Helvetica equivalent  (sans-serif)
  \usepackage{courier}                % Courier New                 (monospace)

  % Convenience macro: switch to Gothic (sans) Korean inside text
  % Usage: \ko{한국어 텍스트}  — switches CJK font to gothic for that span
  \newcommand{\ko}[1]{{\CJKfamily{gr}#1}}

\fi
% ---- End of compiler-dependent block ------------------------------------

% CHAPTER & SECTION TITLE FORMATTING

% Chapter heading style
\titleformat{\chapter}
  [block]
  {\sffamily\Huge\bfseries\centering}
  {\chaptertitlename\ \thechapter.}
  {0.5cm}
  {}

\titleformat{\subsection}{\sffamily\normalsize\bfseries}{\thesubsection.}{1em}{}

\titlespacing*{\subsection}{0cm}{1cm}{0.5cm}
